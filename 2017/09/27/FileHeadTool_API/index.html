getFileheader

public HashMap getFileheader()
用于返回文件头字典

无

返回

HashMap

setFileheader

public void setFileheader(String fileheads,String filesuffix)
用于补充文件头字典

参数

String fileheads,String filesuffix
fileheads为文件头
filesuffix为文件后缀名

返回

无

fileExist

public boolean fileExist(String fileDir)

String fileDir

返回

boolean

fileFormat

public String fileFormat(String fileDir)

String fileDir

返回

String
